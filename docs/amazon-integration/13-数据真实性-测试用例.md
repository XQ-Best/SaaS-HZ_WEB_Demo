### AUTH-01 自动化断言（加入 `audit_amazon_data_coverage.py`）

```python
def audit_synthetic_acos(rows):
    with_acos = [r for r in rows if float(r["acos"] or 0) > 0]
    if len(with_acos) < 5:
        return True
    distinct = len({round(float(r["acos"]), 2) for r in with_acos})
    if distinct == 1:
        return False  # FAIL: 全相同 ACOS
    ratios = []
    for r in with_acos:
        rev = parse_money(r["revenue_30d"])
        spend = parse_money(r["ad_spend_30d"])
        if rev > 0 and spend > 0:
            ratios.append(round(spend / rev, 4))
    if len(ratios) >= 5 and len(set(ratios)) == 1:
        return False  # FAIL: 固定比例 spend
    return True
```
